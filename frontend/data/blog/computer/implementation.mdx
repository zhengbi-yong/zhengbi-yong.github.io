```rust
use tower_http::{

use std::sync::Arc;

#[derive(Clone)]
pub struct AppState {
    pub db: sqlx::PgPool,
    pub redis: deadpool_redis::Pool,
    pub jwt: crate::core::auth::JwtService,
    pub settings: crate::shared::config::Settings,

#[tokio::main]
async fn main() -> anyhow::Result<()> {
    // 初始化 tracing
    tracing_subscriber::registry()
        .with(tracing_subscriber::EnvFilter::from_default_env())
        .with(tracing_subscriber::fmt::layer())
        .init();

    // 加载配置
    let settings = crate::shared::config::Settings::from_env()?;

    // 初始化数据库连接池
    let db = sqlx::PgPool::connect(&settings.database_url).await?;

    // 运行迁移
    sqlx::migrate!("./migrations").run(&db).await?;

    // 初始化 Redis
    let redis = deadpool_redis::Config::from_url(&settings.redis_url)
        .create_pool(Some(deadpool_redis::Runtime::Tokio1))?;

    // 初始化服务
    let jwt = crate::core::auth::JwtService::new(&settings.jwt_secret)?;
    let email_service = crate::core::email::EmailService::new(&settings.smtp)?;

    let state = AppState {
        db,
        redis,
        jwt,
        settings: settings.clone(),
        email_service,
    };

    // 构建路由
    let app = Router::new()
        // 健康检查
        .route("/healthz", axum::routing::get(healthz))
        .route("/readyz", axum::routing::get(readyz))
        // API v1
        .nest("/v1", v1_routes(state.clone()))
        // OpenAPI 文档
        .merge(
            utoipa_swagger_ui::SwaggerUi::new("/swagger-ui")
                .url("/api-docs/openapi.json", crate::api::openapi::openapi_spec()),
        )
        // 中间件
        .layer(TraceLayer::new_for_http())
        .layer(CompressionLayer::new())
        .layer(
            CorsLayer::new()
                .allow_origin(["https://yourdomain.com".parse().unwrap()])
                .allow_methods([
                    axum::http::Method::GET,
                    axum::http::Method::POST,
                    axum::http::Method::PUT,
                    axum::http::Method::DELETE,
                ])
                .allow_headers([
                    axum::http::header::AUTHORIZATION,
                    axum::http::header::ACCEPT,
                ])
                .allow_credentials(true),
        )
        .with_state(state);

    // 启动服务器
    let listener = tokio::net::TcpListener::bind("0.0.0.0:3000").await?;
    tracing::info!("Server listening on {}", listener.local_addr());

    axum::serve(listener, app).await?;

    Ok(())
}

// 健康检查
async fn healthz() -> &'static str {
    "OK"
}

async fn readyz(State(state): State<AppState>) -> Result<(), StatusCode> {
    // 检查数据库连接
    sqlx::query("SELECT 1")
        .fetch_one(&state.db)
        .await
        .map_err(|_| StatusCode::SERVICE_UNAVAILABLE)?;

    // 检查 Redis 连接
    let mut conn = state.redis.get().await
        .map_err(|_| StatusCode::SERVICE_UNAVAILABLE)?;

    redis::cmd("PING")
        .query_async::<_, ()>(&mut conn)
        .await
        .map_err(|_| StatusCode::SERVICE_UNAVAILABLE)?;

    Ok(())
}

// v1 路由定义
fn v1_routes(state: AppState) -> Router<AppState> {
    use axum::routing::{get, post, delete};

    Router::new()

        .route("/auth/register", post(crate::routes::auth::register))
        .route("/auth/login", post(crate::routes::auth::login))
        .route("/auth/refresh", post(crate::routes::auth::refresh))
        .route("/auth/logout", post(crate::routes::auth::logout))
        .route("/auth/me", get(crate::routes::auth::me))

        // 文章相关路由
        .route("/posts/:slug/stats", get(crate::routes::posts::get_stats))
        .route("/posts/:slug/view", post(crate::routes::posts::view))
        .route("/posts/:slug/like", post(crate::routes::posts::like))
        .route("/posts/:slug/like", delete(crate::routes::posts::unlike))

        // 评论相关路由
        .route("/posts/:slug/comments", get(crate::routes::comments::list_comments))
        .route("/posts/:slug/comments", post(crate::routes::comments::create_comment))

        .layer(middleware::from_fn_with_state(
            state.clone(),
            middleware::rate_limit::rate_limit_middleware,
        ))
        .layer(middleware::from_fn_with_state(
            state.clone(),
            middleware::auth::auth_middleware,
        ))
}
```

### 1.4 认证系统实现

#### crates/core/src/auth.rs

```rust
use argon2::{
    password_hash::{rand_core::OsRng, PasswordHash, PasswordHasher, SaltString},
    Argon2,
};
use jsonwebtoken::{decode, encode, DecodingKey, EncodingKey, Header, Validation};
use serde::{Deserialize, Serialize};
use uuid::Uuid;
use chrono::{Duration, Utc};
use sha2::{Digest, Sha256};
use crate::shared::error::AppError;

#[derive(Debug, Serialize, Deserialize)]
pub struct Claims {
    pub sub: String,    // user_id
    pub email: String,
    pub username: String,
    pub exp: i64,       // 过期时间
    pub iat: i64,       // 签发时间
    pub token_type: TokenType,
}

#[derive(Debug, Serialize, Deserialize)]
#[serde(rename_all = "snake_case")]
pub enum TokenType {
    Access,
    Refresh { token_id: Uuid, family_id: Uuid },
}

pub struct JwtService {
    encoding_key: EncodingKey,
    decoding_key: DecodingKey,
}

impl JwtService {
    pub fn new(secret: &str) -> Result<Self, AppError> {
        Ok(Self {
            encoding_key: EncodingKey::from_secret(secret.as_ref()),
            decoding_key: DecodingKey::from_secret(secret.as_ref()),
        })
    }

    // 密码哈希
    pub fn hash_password(&self, password: &str) -> Result<String, AppError> {
        let salt = SaltString::generate(&mut OsRng);
        let argon2 = Argon2::default();
        let password_hash = argon2
            .hash_password(password.as_bytes(), &salt)?
            .to_string();
        Ok(password_hash)
    }

    // 验证密码
    pub fn verify_password(&self, password: &str, hash: &str) -> Result<bool, AppError> {
        let parsed_hash = PasswordHash::new(hash)
            .map_err(|_| AppError::PasswordHashError)?;
        Ok(Argon2::default()
            .verify_password(password.as_bytes(), &parsed_hash)
            .is_ok())
    }

    // 生成 Access Token（15分钟）
    pub fn create_access_token(
        &self,
        user_id: &Uuid,
        email: &str,
        username: &str,
    ) -> Result<String, AppError> {
        let now = Utc::now();
        let claims = Claims {
            sub: user_id.to_string(),
            email: email.to_string(),
            username: username.to_string(),
            exp: (now + Duration::minutes(15)).timestamp(),
            iat: now.timestamp(),
            token_type: TokenType::Access,
        };

        encode(&Header::default(), &claims, &self.encoding_key)
            .map_err(|_| AppError::TokenCreationError)
    }

    // 生成 Refresh Token（7天）
    pub fn create_refresh_token(&self, user_id: &Uuid) -> Result<(String, Uuid), AppError> {
        let token_id = Uuid::new_v4();
        let family_id = Uuid::new_v4();
        let now = Utc::now();

        let claims = Claims {
            sub: user_id.to_string(),
            email: String::new(), // Refresh token 不需要邮箱
            username: String::new(), // Refresh token 不需要用户名
            exp: (now + Duration::days(7)).timestamp(),
            iat: now.timestamp(),
            token_type: TokenType::Refresh { token_id, family_id },
        };

        let token = encode(&Header::default(), &claims, &self.encoding_key)
            .map_err(|_| AppError::TokenCreationError)?;

        Ok((token, family_id))
    }

    // 验证 Access Token
    pub fn verify_access_token(&self, token: &str) -> Result<Claims, AppError> {
        let token_data = decode::<Claims>(
            token,
            &self.decoding_key,
            &Validation::default(),
        )
        .map_err(|_| AppError::InvalidToken)?;

        match token_data.claims.token_type {
            TokenType::Access => Ok(token_data.claims),
            _ => Err(AppError::InvalidTokenType),
        }
    }

    // 验证 Refresh Token
    pub fn verify_refresh_token(&self, token: &str) -> Result<RefreshClaims, AppError> {
        let token_data = decode::<Claims>(
            token,
            &self.decoding_key,
            &Validation::default(),
        )
        .map_err(|_| AppError::InvalidToken)?;

        match token_data.claims.token_type {
            TokenType::Refresh { token_id, family_id } => Ok(RefreshClaims {
                user_id: Uuid::parse_str(&token_data.claims.sub)
                    .map_err(|_| AppError::InvalidUserId)?,
                token_id,
                family_id,
            }),
            _ => Err(AppError::InvalidTokenType),
        }
    }

    // 生成 Token 哈希
    pub fn hash_token(&self, token: &str) -> String {
        let mut hasher = Sha256::new();
        hasher.update(token.as_bytes());
        format!("{:x}", hasher.finalize())
    }
}

#[derive(Debug)]
pub struct RefreshClaims {
    pub user_id: Uuid,
    pub token_id: Uuid,
    pub family_id: Uuid,
}
```

### 1.5 限流中间件实现

#### crates/shared/src/middleware/rate_limit.rs

```rust
use axum::{
    extract::{Request, State},
    http::StatusCode,
    middleware::Next,
    response::Response,
};
use std::time::Duration;
use redis::Script;

// Redis Lua 脚本：固定窗口限流
const RATE_LIMIT_SCRIPT: &str = r#"
local key = KEYS[1]
local limit = tonumber(ARGV[1])
local window = tonumber(ARGV[2])
local current = redis.call("INCR", key)
if current == 1 then
    redis.call("EXPIRE", key, window)
end
if current > limit then
    return 0
end
return 1
"#;

pub async fn rate_limit_middleware<B>(
    State(state): State<crate::api::AppState>,
    request: Request<B>,
    next: Next<B>,
) -> Result<Response, StatusCode> {
    let ip = extract_real_ip(&request);
    let route = extract_route(&request);

    // 构建限流键：rl:{ip}:{route}:{minute_bucket}
    let minute_bucket = chrono::Utc::now().format("%Y%m%d%H%M");
    let key = format!("rl:{}:{}:{}", ip, route, minute_bucket);

    // 根据路由设置不同的限流策略
    let (limit, window) = match route.as_str() {
        "/v1/auth/login" | "/v1/auth/register" => (5, 60),     // 5次/分钟
        "/v1/posts/*/view" => (100, 60),                         // 100次/分钟
        "/v1/posts/*/comments" if request.method().as_str() == "POST" => (10, 60), // 10次/分钟
        _ => (1000, 60),                                         // 默认
    };

    // 执行 Redis 脚本
    let mut conn = state.redis.get().await
        .map_err(|_| StatusCode::INTERNAL_SERVER_ERROR)?;

    let result: i32 = Script::new(RATE_LIMIT_SCRIPT)
        .key(&key)
        .arg(limit)
        .arg(window)
        .invoke_async(&mut conn)
        .await
        .map_err(|_| StatusCode::INTERNAL_SERVER_ERROR)?;

    if result == 0 {
        tracing::warn!("Rate limit exceeded for IP: {}, Route: {}", ip, route);
        return Err(StatusCode::TOO_MANY_REQUESTS);
    }

    Ok(next.run(request).await)
}

fn extract_real_ip<B>(request: &Request<B>) -> String {
    // Cloudflare 优先
    if let Some(ip) = request.headers().get("cf-connecting-ip") {
        return ip.to_str().unwrap_or("unknown").to_string();
    }

    // 其次 x-real-ip
    if let Some(ip) = request.headers().get("x-real-ip") {
        return ip.to_str().unwrap_or("unknown").to_string();
    }

    // 最后 x-forwarded-for (取第一个)
    if let Some(xff) = request.headers().get("x-forwarded-for") {
        return xff.to_str()
            .ok()
            .and_then(|s| s.split(',').next())
            .map(|s| s.trim().to_string())
            .unwrap_or("unknown");
    }

    "unknown".to_string()
}

fn extract_route<B>(request: &Request<B>) -> String {
    // 使用 MatchedPath 获取路由模式
    request
        .extensions()
        .get::<axum::extract::MatchedPath>()
        .map(|m| {
            // 将动态路径段替换为通配符
            m.as_str()
                .split('/')
                .map(|segment| {
                    if segment.chars().all(|c| c.is_alphanumeric() || c == '-' || c == '_') {
                        // 看起来像静态路径
                        segment
                    } else if segment.parse::<uuid::Uuid>().is_ok() {
                        // UUID
                        "*"
                    } else {
                        // 其他动态内容
                        "*"
                    }
                })
                .collect::<Vec<_>>()
                .join("/")
        })
        .unwrap_or_else(|| "unknown".to_string())
}
```

```rust
// 用户注册
pub async fn register(
    State(state): State<AppState>,
    Json(payload): Json<RegisterRequest>,
) -> Result<Json<AuthResponse>, AppError> {
    // 验证输入
    if payload.email.len() < 3 || payload.username.len() < 3 || payload.password.len() < 8 {
        return Err(AppError::InvalidInput);
    }

    // 检查邮箱是否已存在
    let existing = sqlx::query_scalar!(
        "SELECT EXISTS(SELECT 1 FROM users WHERE email = $1)",
        payload.email
    )
    .fetch_one(&state.db)
    .await?;

    if existing {
        return Err(AppError::EmailAlreadyExists);
    }

    // 检查用户名是否已存在
    let existing = sqlx::query_scalar!(
        "SELECT EXISTS(SELECT 1 FROM users WHERE username = $1)",
        payload.username
    )
    .fetch_one(&state.db)
    .await?;

    if existing {
        return Err(AppError::UsernameAlreadyExists);
    }

    // 创建用户
    let password_hash = state.jwt.hash_password(&payload.password)?;

    let user = sqlx::query_as!(
        crate::db::models::User,
        r#"
        INSERT INTO users (email, username, password_hash)
        VALUES ($1, $2, $3)
        RETURNING *
        "#,
        payload.email,
        payload.username,
        password_hash
    )
    .fetch_one(&state.db)
    .await?;

        user.id,
        token_hash,
        family_id,
        "127.0.0.1" // TODO: 从请求中获取真实 IP

// 刷新 token
pub async fn refresh(
    State(state): State<AppState>,
    jar: axum::extract::CookieJar,
) -> Result<Response, AppError> {
    let refresh_token = jar
        .get("refresh_token")
        .ok_or(AppError::MissingRefreshToken)?
        .value();

    // 验证 refresh token
    let claims = state.jwt.verify_refresh_token(&refresh_token)?;

    // 检查 token 是否有效且未被撤销
    let token_hash = state.jwt.hash_token(&refresh_token);
    let token_record = sqlx::query!(
        "SELECT * FROM refresh_tokens WHERE token_hash = $1 AND revoked_at IS NULL",
        token_hash
    )
    .fetch_optional(&state.db)
    .await?
    .ok_or(AppError::InvalidToken)?;

    // 获取用户信息
    let user = sqlx::query_as!(
        crate::db::models::User,
        "SELECT * FROM users WHERE id = $1",
        claims.user_id
    )
    .fetch_one(&state.db)
    .await?;

    // 生成新的 access token
    let access_token = state.jwt.create_access_token(&user.id, &user.email, &user.username)?;

    // Token Rotation：生成新的 refresh token
    let (new_refresh_token, new_family_id) = state.jwt.create_refresh_token(&user.id)?;

    // 吊销同族的所有旧 token
    sqlx::query!(
        "UPDATE refresh_tokens SET revoked_at = NOW() WHERE family_id = $1",
        claims.family_id
    )
    .execute(&state.db)
    .await?;

    // 存储新的 refresh token
    let new_token_hash = state.jwt.hash_token(&new_refresh_token);
    sqlx::query!(
        r#"
        INSERT INTO refresh_tokens (user_id, token_hash, family_id, expires_at, replaced_by_hash)
        VALUES ($1, $2, $3, NOW() + INTERVAL '7 days', $4)
        "#,
        user.id,
        new_token_hash,
        new_family_id,
        token_hash
    )
    .execute(&state.db)
    .await?;

    // 更新 cookie
    let cookie = format!(
        "refresh_token={}; HttpOnly; Secure; SameSite=Lax; Path=/v1/auth/refresh; Max-Age={}",
        new_refresh_token,
        7 * 24 * 60 * 60
    );

    let response = Json(serde_json::json!({
        "access_token": access_token,
    }));

    Ok((
        [(header::SET_COOKIE, cookie)],
        response.into_response(),
    ).into_response())
}
```

```rust
// 获取文章统计（可缓存）
pub async fn get_stats(
    State(state): State<AppState>,
    Path(slug): Path<String>,
) -> Result<impl IntoResponse, AppError> {
    // 尝试从 Redis 缓存获取
    let cache_key = format!("post_stats:{}", slug);

    let mut conn = state.redis.get().await?;

    if let Ok(Some(cached)) = redis::cmd("GET")
        .arg(&cache_key)
        .query_async::<_, Option<String>>(&mut conn)
        .await
    {
        if let Ok(stats) = serde_json::from_str::<PostStats>(&cached) {
            return Ok((
                [
                    (header::CACHE_CONTROL, "public, s-maxage=5, stale-while-revalidate=60"),
                    (header::ETAG, format!("\"{}\"", compute_etag(&stats))),
                ],
                Json(stats),
            ).into_response());
        }
    }

    // 从数据库获取
    let stats = sqlx::query_as!(
        PostStats,
        "SELECT * FROM post_stats WHERE post_slug = $1",
        slug
    )
    .fetch_optional(&state.db)
    .await?;

    if let Some(stats) = stats {
        // 缓存 5 秒
        if let Ok(json) = serde_json::to_string(&stats) {
            let _: () = redis::cmd("SETEX")
                .arg(&cache_key)
                .arg(5)
                .arg(&json)
                .query_async(&mut conn)
                .await?;
        }

        Ok((
            [
                (header::CACHE_CONTROL, "public, s-maxage=5, stale-while-revalidate=60"),
                (header::ETAG, format!("\"{}\"", compute_etag(&stats))),
            ],
            Json(stats),
        ).into_response())
    } else {
        // 如果统计不存在，创建默认值
        let default_stats = PostStats {
            slug: slug.clone(),
            view_count: 0,
            like_count: 0,
            comment_count: 0,
        };

        sqlx::query!(
            "INSERT INTO post_stats (post_slug) VALUES ($1)",
            slug
        )
        .execute(&state.db)
        .await?;

        Ok(Json(default_stats)).map_err(Into::into)
    }
}

// 记录浏览（可匿名，强限流）
pub async fn view(
    State(state): State<AppState>,
    Path(slug): Path<String>,
) -> Result<(), AppError> {

    let exists = sqlx::query_scalar!(
        "SELECT EXISTS(SELECT 1 FROM post_stats WHERE post_slug = $1)",
        slug
    )
    .fetch_one(&state.db)
    .await?;

    if !exists {
        return Err(AppError::PostNotFound);
    }

    // Redis 计数
    let mut conn = state.redis.get().await?;
    let pv_key = format!("pv:{}", slug);
    let count: i64 = redis::cmd("INCR")
        .arg(&pv_key)
        .query_async(&mut conn)
        .await?;

    // 每 10 次浏览写入一次 outbox
    if count % 10 == 0 {
        sqlx::query!(
            r#"
            INSERT INTO outbox_events (topic, payload, run_after)
            VALUES ('post.viewed', $1, NOW())
            "#,
            serde_json::json!({
                "post_slug": slug,
                "count": count,
            })
        )
        .execute(&state.db)
        .await?;
    }

    Ok(())
}

// 点赞（需要登录）
pub async fn like(
    State(state): State<AppState>,
    auth_user: AuthUser,
    Path(slug): Path<String>,
) -> Result<(), AppError> {
    // 使用事务确保原子性
    let mut tx = state.db.begin().await?;

    // 检查文章是否存在
    let exists = sqlx::query_scalar!(
        "SELECT EXISTS(SELECT 1 FROM post_stats WHERE post_slug = $1)",
        slug
    )
    .fetch_one(&mut *tx)
    .await?;

    if !exists {
        tx.rollback().await?;
        return Err(AppError::PostNotFound);
    }

    // 检查是否已点赞
    let already_liked = sqlx::query_scalar!(
        "SELECT EXISTS(SELECT 1 FROM post_likes WHERE post_slug = $1 AND user_id = $2)",
        slug,
        auth_user.id
    )
    .fetch_one(&mut *tx)
    .await?;

    if already_liked {
        tx.rollback().await?;
        return Err(AppError::AlreadyLiked);
    }

    // 添加点赞记录
    sqlx::query!(
        "INSERT INTO post_likes (post_slug, user_id) VALUES ($1, $2)",
        slug,
        auth_user.id
    )
    .execute(&mut *tx)
    .await?;

    // 更新计数
    sqlx::query!(
        "UPDATE post_stats SET like_count = like_count + 1 WHERE post_slug = $1",
        slug
    )
    .execute(&mut *tx)
    .await?;

    tx.commit().await?;

    // 清除缓存
    let mut conn = state.redis.get().await?;
    let cache_key = format!("post_stats:{}", slug);
    let _: () = redis::cmd("DEL")
        .arg(&cache_key)
        .query_async(&mut conn)
        .await?;

    Ok(())
}

// 取消点赞
pub async fn unlike(

    sqlx::query!(

        slug
    )
    .execute(&mut *tx)
    .await?;

    tx.commit().await?;

    // 清除缓存
    let mut conn = state.redis.get().await?;
    let cache_key = format!("post_stats:{}", slug);

// 计算 ETag
fn compute_etag(stats: &PostStats) -> String {
    use std::hash::{Hash, Hasher};
    let mut hasher = std::collections::hash_map::DefaultHasher::new();
    stats.view_count.hash(&mut hasher);
    stats.like_count.hash(&mut hasher);
    stats.comment_count.hash(&mut hasher);
    format!("{:x}", hasher.finish())
}
```

```rust
// HTML 清理配置
fn sanitize_comment_html(input: &str) -> String {
    Builder::default()
        .add_tags(&["p", "br", "strong", "em", "code", "pre", "a", "blockquote"])
        .add_tag_attributes("a", &["href", "title"])
        .add_tag_attributes("pre", &["class"])
        .add_tag_attributes("code", &["class"])
        .link_rel(Some("nofollow ugc"))
        .url_relative(ammonia::UrlRelative::PassThrough)
        .clean(input)
        .to_string()
}

// 创建评论
pub async fn create_comment(
    State(state): State<AppState>,
    auth_user: AuthUser,
    Path(slug): Path<String>,
    Json(payload): Json<CreateCommentRequest>,
) -> Result<Json<CommentResponse>, AppError> {
    // 检查文章是否存在
    let exists = sqlx::query_scalar!(
        "SELECT EXISTS(SELECT 1 FROM post_stats WHERE post_slug = $1)",
        slug
    )
    .fetch_one(&state.db)
    .await?;

    if !exists {
        return Err(AppError::PostNotFound);
    }

    // 验证内容
    if payload.content.trim().is_empty() {
        return Err(AppError::EmptyComment);
    }

    if payload.content.len() > 2000 {
        return Err(AppError::CommentTooLong);
    }

    // 清理 HTML
    let html_content = sanitize_comment_html(&payload.content);

    // 计算路径和深度
    let (path, depth) = if let Some(parent_id) = payload.parent_id {
        let parent = sqlx::query!(
            "SELECT path, depth FROM comments WHERE id = $1",
            parent_id
        )
        .fetch_optional(&state.db)
        .await?
        .ok_or(AppError::CommentNotFound)?;

        if parent.depth >= 5 {
            return Err(AppError::CommentTooDeep);
        }

        let new_comment_id = Uuid::new_v4().simple().to_string();
        let path = format!("{}.c{}", parent.path, new_comment_id);
        (path, parent.depth + 1)
    } else {
        let new_comment_id = Uuid::new_v4().simple().to_string();
        (format!("c{}", new_comment_id), 0)
    };

    // 创建评论
    let comment = sqlx::query_as!(
        crate::db::models::Comment,
        r#"
        INSERT INTO comments (
            post_slug, user_id, parent_id, content, html_sanitized,
            path, depth, status, created_ip, user_agent
        ) VALUES ($1, $2, $3, $4, $5, $6, $7, 'pending', $8, $9)
        RETURNING *
        "#,
        slug,
        auth_user.id,
        payload.parent_id,
        payload.content,
        html_content,
        path,
        depth,
        "127.0.0.1", // TODO: 从请求中获取真实 IP
        "unknown"     // TODO: 从请求中获取 User-Agent
    )
    .fetch_one(&state.db)
    .await?;

    // 更新评论计数
    sqlx::query!(
        "UPDATE post_stats SET comment_count = comment_count + 1 WHERE post_slug = $1",
        slug
    )
    .execute(&state.db)
    .await?;

    // 写入 Outbox 供 Worker 处理
    sqlx::query!(
        r#"
        INSERT INTO outbox_events (topic, payload)
        VALUES ('comment.created', $1)
        "#,
        serde_json::json!({
            "comment_id": comment.id,
            "post_slug": slug,
            "user_id": auth_user.id,
        })
    )
    .execute(&state.db)
    .await?;

    // 获取用户信息
    let user = sqlx::query!(
        "SELECT username, profile FROM users WHERE id = $1",
        auth_user.id
    )
    .fetch_one(&state.db)
    .await?;

    Ok(Json(CommentResponse {
        id: comment.id,
        content: comment.content,
        html_sanitized: comment.html_sanitized,
        user: CommentUser {
            username: user.username,
            profile: user.profile,
        },
        created_at: comment.created_at,
        like_count: comment.like_count,
        replies: vec![], // 创建时不包含回复
    }))
}

// 获取评论列表（分页 + 游标）
pub async fn list_comments(
    State(state): State<AppState>,
    Path(slug): Path<String>,
    Query(params): Query<CommentListParams>,
) -> Result<Json<CommentListResponse>, AppError> {
    let limit = params.limit.unwrap_or(20).min(100);

    // 只返回已审核的评论
    let comments = if let Some(cursor) = params.cursor {
        let cursor_uuid = Uuid::parse_str(&cursor)
            .map_err(|_| AppError::InvalidCursor)?;

        sqlx::query_as!(
            crate::db::models::CommentWithUser,
            r#"
            WITH cte AS (
                SELECT c.*
                FROM comments c
                WHERE c.post_slug = $1
                    AND c.status = 'approved'
                    AND c.created_at < (SELECT created_at FROM comments WHERE id = $2)
                ORDER BY c.created_at DESC
                LIMIT $3
            )
            SELECT
                cte.*,
                u.username,
                u.profile as "profile: _"
            FROM cte
            JOIN users u ON cte.user_id = u.id
            "#,
            slug,
            cursor_uuid,
            limit as i64
        )
        .fetch_all(&state.db)
        .await?
    } else {
        sqlx::query_as!(
            crate::db::models::CommentWithUser,
            r#"
            SELECT
                c.*,
                u.username,
                u.profile as "profile: _"
            FROM comments c
            JOIN users u ON c.user_id = u.id
            WHERE c.post_slug = $1 AND c.status = 'approved'
            ORDER BY c.created_at DESC
            LIMIT $2
            "#,
            slug,
            limit as i64
        )
        .fetch_all(&state.db)
        .await?
    };

    let next_cursor = comments
        .last()
        .map(|c| c.id.to_string());

    // 转换为响应格式
    let comment_responses: Vec<CommentResponse> = comments
        .into_iter()
        .map(|c| CommentResponse {
            id: c.id,
            content: c.content,
            html_sanitized: c.html_sanitized,
            user: CommentUser {
                username: c.username,
                profile: c.profile,
            },
            created_at: c.created_at,
            like_count: c.like_count,
            replies: vec![], // 暂时不加载嵌套回复
        })
        .collect();

    Ok(Json(CommentListResponse {
        comments: comment_responses,
        next_cursor,
    }))
}
```

```rust
use sqlx::PgPool;
use serde_json::Value;
use std::time::Duration;

#[tokio::main]
async fn main() -> anyhow::Result<()> {
    // 初始化日志
    tracing_subscriber::registry()
        .with(tracing_subscriber::EnvFilter::from_default_env())
        .with(tracing_subscriber::fmt::layer())
        .init();

    // 加载配置
    let settings = crate::shared::config::Settings::from_env()?;

    // 初始化数据库
    let db = sqlx::PgPool::connect(&settings.database_url).await?;

    // 初始化 Redis
    let redis = deadpool_redis::Config::from_url(&settings.redis_url)
        .create_pool(Some(deadpool_redis::Runtime::Tokio1))?;

    tracing::info!("Worker started");

    let mut interval = tokio::time::interval(Duration::from_secs(5));

    loop {
        tokio::select! {
            _ = interval.tick() => {
                // 批量处理 Outbox 事件
                if let Err(e) = process_outbox_events(&db).await {
                    tracing::error!("Error processing outbox events: {}", e);
                }

                // 批量刷新浏览量
                if let Err(e) = flush_view_counts(&db, &redis).await {
                    tracing::error!("Error flushing view counts: {}", e);
                }
            }
        }
    }
}

async fn process_outbox_events(db: &PgPool) -> anyhow::Result<()> {
    let events = sqlx::query_as!(
        crate::db::models::OutboxEvent,
        r#"
        UPDATE outbox_events
        SET status = 'processing'
        WHERE id = (
            SELECT id FROM outbox_events
            WHERE status = 'pending'
                AND run_after <= NOW()
            ORDER BY created_at
            FOR UPDATE SKIP LOCKED
            LIMIT 100
        )
        RETURNING *
        "#
    )
    .fetch_all(db)
    .await?;

    for event in events {
        match event.topic.as_str() {
            "comment.created" => {
                if let Err(e) = handle_comment_created(db, &event.payload).await {
                    tracing::error!("Error handling comment.created: {}", e);
                    mark_event_failed(db, &event.id, &e.to_string()).await?;
                }
            }
            "post.viewed" => {
                // 浏览量已经在 flush_view_counts 中处理
            }
            _ => {
                tracing::warn!("Unknown event topic: {}", event.topic);
            }
        }

        // 标记为完成
        sqlx::query!(
            "UPDATE outbox_events SET status = 'done' WHERE id = $1",
            event.id
        )
        .execute(db)
        .await?;
    }

    Ok(())
}

async fn handle_comment_created(db: &PgPool, payload: &Value) -> anyhow::Result<()> {
    let comment_id: Uuid = serde_json::from_value(payload["comment_id"].clone())?;
    let post_slug: String = serde_json::from_value(payload["post_slug"].clone())?;

    // TODO: 发送邮件通知给文章作者
    // TODO: 如果是回复，发送邮件通知给被回复的用户

    tracing::info!(
        "New comment {} created on post {}",
        comment_id,
        post_slug
    );

    Ok(())
}

async fn flush_view_counts(db: &PgPool, redis: &deadpool_redis::Pool) -> anyhow::Result<()> {
    let mut conn = redis.get().await?;

    // 获取所有浏览量计数
    let keys: Vec<String> = redis::cmd("KEYS")
        .arg("pv:*")
        .query_async(&mut conn)
        .await?;

    if keys.is_empty() {
        return Ok(());
    }

    // 批量获取计数
    let values: Vec<Option<i64>> = redis::cmd("MGET")
        .arg(&keys)
        .query_async(&mut conn)
        .await?;

    // 构建批量更新
    let mut updates = Vec::new();
    for (key, value) in keys.iter().zip(values) {
        if let Some(count) = value {
            if let Some(slug) = key.strip_prefix("pv:") {
                updates.push((slug, count));
            }
        }
    }

    // 使用 VALUES 子句批量更新
    if !updates.is_empty() {
        let mut query = String::from(
            r#"
            UPDATE post_stats
            SET view_count = view_count + update_data.count
            FROM (VALUES
            "#
        );

        for (i, (slug, count)) in updates.iter().enumerate() {
            if i > 0 {
                query.push_str(", ");
            }
            query.push_str(&format!("('${}', {})", slug, count));
        }

        query.push_str(
            r#"
            ) AS update_data(slug, count)
            WHERE post_stats.post_slug = update_data.slug
            "#
        );

        sqlx::query(&query).execute(db).await?;

        // 清理已处理的计数
        let _: () = redis::cmd("DEL")
            .arg(&keys)
            .query_async(&mut conn)
            .await?;
    }

    Ok(())
}

async fn mark_event_failed(db: &PgPool, event_id: &Uuid, error: &str) -> anyhow::Result<()> {
    sqlx::query!(
        r#"
        UPDATE outbox_events
        SET status = 'failed',
            last_error = $2,
            attempts = attempts + 1,
            run_after = NOW() + INTERVAL '5 minutes'
        WHERE id = $1
        "#,
        event_id,
        error
    )
    .execute(db)
    .await?;

    Ok(())
}
```

```typescript
import axios, { AxiosInstance, AxiosResponse } from 'axios'
import { getCookie, deleteCookie } from 'cookies-next'

const API_BASE_URL = process.env.NEXT_PUBLIC_API_URL || 'https://api.yourdomain.com/v1'

class ApiClient {
  private axios: AxiosInstance

  constructor() {
    this.axios = axios.create({
      baseURL: API_BASE_URL,
      timeout: 10000,
    })

    // Request interceptor - 添加认证头
    this.axios.interceptors.request.use((config) => {
      const token = window.localStorage.getItem('access_token')
      if (token) {
        config.headers.Authorization = `Bearer ${token}`
      }
      return config
    })

    // Response interceptor - 处理 401 和 token 刷新
    this.axios.interceptors.response.use(
      (response: AxiosResponse) => response,
      async (error) => {
        const originalRequest = error.config

        // 如果是 401 错误且不是刷新请求
        if (error.response?.status === 401 && !originalRequest._retry) {
          originalRequest._retry = true

          try {
            await this.refreshToken()
            // 重试原始请求
            const token = window.localStorage.getItem('access_token')
            if (token) {
              originalRequest.headers.Authorization = `Bearer ${token}`
            }
            return this.axios(originalRequest)
          } catch (refreshError) {
            // 刷新失败，清除认证信息并重定向到登录页
            window.localStorage.removeItem('access_token')
            window.location.href = '/login'
            return Promise.reject(refreshError)
          }
        }

        return Promise.reject(error)
      }
    )
  }

  private async refreshToken(): Promise<void> {
    const response = await axios.post(
      `${API_BASE_URL}/auth/refresh`,
      {},
      {
        withCredentials: true,
      }
    )
    const { access_token } = response.data
    window.localStorage.setItem('access_token', access_token)
  }

  // 认证相关方法
  async register(data: { email: string; username: string; password: string }) {
    return this.axios.post('/auth/register', data)
  }

  async login(data: { email: string; password: string }) {
    const response = await this.axios.post('/auth/login', data, {
      withCredentials: true,
    })
    return response.data
  }

  async logout() {
    try {
      await this.axios.post('/auth/logout', {}, { withCredentials: true })
    } finally {
      window.localStorage.removeItem('access_token')
      deleteCookie('refresh_token')
    }
  }

  async me() {
    return this.axios.get('/auth/me')
  }

  // 文章相关方法
  async getPostStats(slug: string) {
    return this.axios.get(`/posts/${slug}/stats`)
  }

  async viewPost(slug: string) {
    return this.axios.post(`/posts/${slug}/view`)
  }

  async likePost(slug: string) {
    return this.axios.post(`/posts/${slug}/like`)
  }

  async unlikePost(slug: string) {
    return this.axios.delete(`/posts/${slug}/like`)
  }

  // 评论相关方法
  async getComments(
    slug: string,
    params?: {
      cursor?: string
      limit?: number
    }
  ) {
    return this.axios.get(`/posts/${slug}/comments`, { params })
  }

  async createComment(
    slug: string,
    data: {
      parent_id?: string
      content: string
    }
  ) {
    return this.axios.post(`/posts/${slug}/comments`, data)
  }

  async likeComment(commentId: string) {
    return this.axios.post(`/comments/${commentId}/like`)
  }
}

export const api = new ApiClient()
```
