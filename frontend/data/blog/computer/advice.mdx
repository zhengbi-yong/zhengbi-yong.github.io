# Rust 高性能前后端分离博客平台架构方案（终极落地版）

## 项目现状分析

当前博客项目是一个基于 Next.js 16 的静态站点，使用 Contentlayer2 处理 MDX 内容。为实现极致性能和最高安全性的用户交互功能，将演进为前后端分离架构，后端采用 Rust 实现。

## 一、最终架构设计

### 1.1 推荐拓扑（最稳最省心）

```
浏览器
  │  静态页面/资源
  ▼
Cloudflare CDN/Edge（缓存/WAF/基础限流）
  │  /api/* 动态请求（带 WAF、Bot 管控）
  ▼
K8s Ingress（Nginx 或 Envoy，TLS 终止可在 CF 或 Ingress）
  ▼
Rust API（Axum）
  │        │
  │        ├── Redis（缓存、计数、分布式限流、会话）
  │        └── PostgreSQL（强一致：用户、评论、点赞等）
  │
  └── Worker（Rust 后台任务：刷计数、发邮件、索引等）
```

**核心原则**：

- 静态内容（MDX）继续走 Next.js + CDN，不塞 DB
- DB 只存"动态域"：用户、评论、点赞、订阅、统计
- 计数类优先 Redis 聚合 + 异步落库
- 鉴权用短 Access Token + Refresh Token 轮换
- 限流两层：Cloudflare/Ingress + Redis 原子限流

```sql
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id UUID NOT NULL REFERENCES users(id) ON DELETE CASCADE,

  family_id UUID NOT NULL,                -- 同一设备/会话族，用于轮换与异常检测

  expires_at TIMESTAMPTZ NOT NULL,

  last_used_at TIMESTAMPTZ,
  user_agent TEXT,
  ip INET,
  UNIQUE(user_id, token_hash)
);

CREATE INDEX idx_refresh_tokens_user ON refresh_tokens(user_id);
```

```rust
    Json(payload): Json<RegisterRequest>,
) -> Result<Json<AuthResponse>, AppError> {
    // TODO: 添加 Cloudflare Turnstile 验证

// 登录
pub async fn login(
    State(state): State<AppState>,
    Json(payload): Json<LoginRequest>,
) -> Result<Response, AppError> {
    // 查找并验证用户
    let user = find_user_by_email(&state.db, &payload.email).await?;
    if !state.jwt.verify_password(&payload.password, &user.password_hash)? {
        return Err(AppError::InvalidCredentials);
    }

    // 生成 token
    let access_token = state.jwt.create_access_token(&user.id, &user.email, &user.username)?;
    let (refresh_token, family_id) = state.jwt.create_refresh_token(&user.id)?;

    // 存储 refresh token
    store_refresh_token(&state.db, &user.id, &refresh_token, &family_id).await?;

    // 设置 HttpOnly cookie
    let cookie = format!(
        "refresh_token={}; HttpOnly; Secure; SameSite=Lax; Path=/v1/auth/refresh; Max-Age={}",
        refresh_token,
        7 * 24 * 60 * 60 // 7天
    );

    let response = Json(json!({
        "access_token": access_token,
        "user": user.into(),
    }));

    Ok((
        [(header::SET_COOKIE, cookie)],
        response.into_response(),
    ).into_response())
}

// 刷新 token
pub async fn refresh(
    State(state): State<AppState>,
    jar: CookieJar,
) -> Result<Response, AppError> {
    let refresh_token = jar
        .get("refresh_token")
        .ok_or(AppError::MissingRefreshToken)?
        .value();

    // 验证 refresh token
    let claims = state.jwt.verify_refresh_token(&refresh_token)?;

    // 检查 token 是否被撤销
    if !is_refresh_token_valid(&state.db, &claims.token_id).await? {
        return Err(AppError::InvalidToken);
    }

    // 获取用户信息
    let user = find_user_by_id(&state.db, &claims.user_id).await?;

    // 生成新的 access token
    let access_token = state.jwt.create_access_token(&user.id, &user.email, &user.username)?;

    // Token Rotation：生成新的 refresh token
    let (new_refresh_token, new_family_id) = state.jwt.create_refresh_token(&user.id)?;

    // 吊销旧 token，存储新 token
    revoke_family_tokens(&state.db, &claims.family_id).await?;
    store_refresh_token(&state.db, &user.id, &new_refresh_token, &new_family_id).await?;

    // 更新 cookie
    let cookie = format!(
        "refresh_token={}; HttpOnly; Secure; SameSite=Lax; Path=/v1/auth/refresh; Max-Age={}",
        new_refresh_token,
        7 * 24 * 60 * 60
    );

    let response = Json(json!({
        "access_token": access_token,
    }));

    Ok((
        [(header::SET_COOKIE, cookie)],
        response.into_response(),
    ).into_response())
}
```

```rust
// routes/posts.rs
use axum::{
    extract::{Path, State},
    response::{IntoResponse, Json},
    middleware,

use tower_http::compression::CompressionLayer;

// 获取文章统计（可缓存）
pub async fn get_stats(
    State(state): State<AppState>,
    Path(slug): Path<String>,
) -> Result<impl IntoResponse, AppError> {
    let stats = state.cache
        .get_or_set(
            &format!("post_stats:{}", slug),
            || async {
                fetch_post_stats(&state.db, &slug).await
            },
            Some(Duration::from_secs(5)),
        )
        .await?;

    Ok((
        [
            (header::CACHE_CONTROL, "public, s-maxage=5, stale-while-revalidate=60"),
            (header::ETAG, format!("\"{}\"", compute_etag(&stats))),
        ],
        Json(stats),
    ))
}

// 浏览量（可匿名，强限流）
pub async fn view(
    State(state): State<AppState>,
    Path(slug): Path<String>,
) -> Result<(), AppError> {
    // 两层限流检查
    check_rate_limit(&state, "view", &slug).await?;

    // Redis 计数
    let count = state.redis
        .incr(&format!("pv:{}", slug))
        .await?;

    // 异步写入 Outbox（可选）
    if count % 10 == 0 {
        write_outbox_event(
            &state.db,
            "post.viewed",
            json!({
                "post_slug": slug,
                "count": count,
            }),
        ).await?;
    }

    Ok(())
}

// 点赞（需要登录）
pub async fn like(
    State(state): State<AppState>,
    auth_user: AuthUser,
    Path(slug): Path<String>,
) -> Result<(), AppError> {
    // 使用事务确保原子性
    let mut tx = state.db.begin().await?;

    // 检查是否已点赞
    let exists = sqlx::query_scalar!(
        "SELECT EXISTS(SELECT 1 FROM post_likes WHERE post_slug = $1 AND user_id = $2)",
        slug,
        auth_user.id
    )
    .fetch_one(&mut *tx)
    .await?;

    if exists {
        return Err(AppError::AlreadyLiked);
    }

    // 添加点赞记录
    sqlx::query!(
        "INSERT INTO post_likes (post_slug, user_id) VALUES ($1, $2)",
        slug,
        auth_user.id
    )
    .execute(&mut *tx)
    .await?;

    // 更新计数
    sqlx::query!(
        "UPDATE post_stats SET like_count = like_count + 1 WHERE post_slug = $1",
        slug
    )
    .execute(&mut *tx)
    .await?;

    tx.commit().await?;

    // 更新缓存
    state.redis.del(&format!("post_stats:{}", slug)).await?;

    Ok(())
}
```

```rust
// routes/comments.rs
use ammonia::Builder;

// 创建评论
pub async fn create_comment(
    State(state): State<AppState>,
    auth_user: AuthUser,
    Path(slug): Path<String>,
    Json(payload): Json<CreateCommentRequest>,
) -> Result<Json<CommentResponse>, AppError> {
    // 检查限流
    check_rate_limit(&state, "comment", &auth_user.id.to_string()).await?;

    // 生成路径
    let path = if let Some(parent_id) = payload.parent_id {
        let parent = find_comment(&state.db, &parent_id).await?;
        format!("{}.c{}", parent.path, uuid::Uuid::new_v4().simple().to_string())
    } else {
        format!("c{}", uuid::Uuid::new_v4().simple().to_string())
    };

    // 清理 HTML
    let html_content = sanitize_comment_html(&payload.content);

    // 创建评论
    let comment = sqlx::query_as!(
        Comment,
        r#"
        INSERT INTO comments (
            post_slug, user_id, parent_id, content, html_sanitized,
            path, depth, status, created_ip, user_agent
        ) VALUES ($1, $2, $3, $4, $5, $6, $7, 'pending', $8, $9)
        RETURNING *
        "#,
        slug,
        auth_user.id,
        payload.parent_id,
        payload.content,
        html_content,
        path,
        path.split('.').count() as i32 - 1,
        extract_real_ip(),
        extract_user_agent()
    )
    .fetch_one(&state.db)
    .await?;

    // 更新评论计数
    sqlx::query!(
        "UPDATE post_stats SET comment_count = comment_count + 1 WHERE post_slug = $1",
        slug
    )
    .execute(&state.db)
    .await?;

    // 写入 Outbox 供 Worker 处理（发送通知等）
    write_outbox_event(
        &state.db,
        "comment.created",
        json!({
            "comment_id": comment.id,
            "post_slug": slug,
        }),
    ).await?;

    Ok(Json(comment.into()))
}

// 获取评论列表（分页 + 游标）
pub async fn list_comments(
    State(state): State<AppState>,
    Path(slug): Path<String>,
    Query(params): Query<CommentListParams>,
) -> Result<Json<CommentListResponse>, AppError> {
    let limit = params.limit.unwrap_or(20).min(100);

    let comments = if let Some(cursor) = params.cursor {
        sqlx::query_as!(
            CommentWithUser,
            r#"
            SELECT
                c.*,
                u.username,
                u.profile as "profile: _"
            FROM comments c
            JOIN users u ON c.user_id = u.id
            WHERE c.post_slug = $1
                AND c.status = 'approved'
                AND c.created_at < (SELECT created_at FROM comments WHERE id = $2)
            ORDER BY c.created_at DESC
            LIMIT $3
            "#,
            slug,
            cursor,
            limit as i64
        )
        .fetch_all(&state.db)
        .await?
    } else {
        sqlx::query_as!(
            CommentWithUser,
            r#"
            SELECT
                c.*,
                u.username,
                u.profile as "profile: _"
            FROM comments c
            JOIN users u ON c.user_id = u.id
            WHERE c.post_slug = $1 AND c.status = 'approved'
            ORDER BY c.created_at DESC
            LIMIT $2
            "#,
            slug,
            limit as i64
        )
        .fetch_all(&state.db)
        .await?
    };

    let next_cursor = comments
        .last()
        .map(|c| c.id.to_string());

    Ok(Json(CommentListResponse {
        comments: comments.into_iter().map(|c| c.into()).collect(),
        next_cursor,
    }))
}

// HTML 清理工具
fn sanitize_comment_html(input: &str) -> String {
    Builder::default()
        .add_tags(&["p", "br", "strong", "em", "code", "pre", "a"])
        .add_tag_attributes("a", &["href", "title"])
        .link_rel(Some("nofollow ugc"))
        .url_relative(ammonia::UrlRelative::PassThrough)
        .clean(input)
        .to_string()
}
```

### 4.3 应用状态与路由

```rust
// crates/api/src/main.rs
use axum::{Router, middleware};
use tower_http::{
    trace::TraceLayer,
    compression::CompressionLayer,
    cors::CorsLayer,
};
use std::sync::Arc;

#[derive(Clone)]
pub struct AppState {
    pub db: sqlx::PgPool,
    pub redis: deadpool_redis::Pool,
    pub jwt: crate::core::auth::JwtService,
    pub settings: crate::shared::config::Settings,
    pub email_service: crate::core::email::EmailService,
}

#[tokio::main]
async fn main() -> anyhow::Result<()> {
    // 初始化 tracing
    tracing_subscriber::registry()
        .with(tracing_subscriber::EnvFilter::from_default_env())
        .with(tracing_subscriber::fmt::layer())
        .init();

    // 加载配置
    let settings = crate::shared::config::Settings::from_env()?;

    // 初始化数据库
    let db = sqlx::PgPool::connect(&settings.database_url).await?;

    // 初始化 Redis
    let redis = deadpool_redis::Config::from_url(&settings.redis_url)
        .create_pool(Some(deadpool_redis::Runtime::Tokio1))?;

    // 初始化服务
    let jwt = crate::core::auth::JwtService::new(&settings.jwt_secret)?;
    let email_service = crate::core::email::EmailService::new(&settings.smtp)?;

    let state = AppState {
        db,
        redis,
        jwt,
        settings: settings.clone(),
        email_service,
    };

    // 构建路由
    let app = Router::new()
        // 健康检查
        .route("/healthz", axum::routing::get(healthz))
        .route("/readyz", axum::routing::get(readyz))
        // API v1
        .nest("/v1", v1_routes(state.clone()))
        // OpenAPI 文档
        .merge(
            utoipa_swagger_ui::SwaggerUi::new("/swagger-ui")
                .url("/api-docs/openapi.json", crate::api::openapi::openapi_spec()),
        )
        // 中间件
        .layer(
            TraceLayer::new_for_http()
                .make_span_with(|request: &axum::http::Request<_>| {
                    tracing::info_span!(
                        "http_request",
                        method = %request.method(),
                        uri = %request.uri(),
                        version = ?request.version(),
                    )
                }),
        )
        .layer(CompressionLayer::new())
        .layer(
            CorsLayer::new()
                .allow_origin(["https://yourdomain.com".parse::<HeaderValue>().unwrap()])
                .allow_methods([Method::GET, Method::POST, Method::PUT, Method::DELETE])
                .allow_headers([header::AUTHORIZATION, header::ACCEPT])
                .allow_credentials(true),
        )
        .with_state(state);

    // 启动服务器
    let listener = tokio::net::TcpListener::bind("0.0.0.0:3000").await?;
    tracing::info!("Server listening on {}", listener.local_addr());

    axum::serve(listener, app).await?;

    Ok(())
}

// v1 路由
fn v1_routes(state: AppState) -> Router<AppState> {
    use axum::routing::{get, post, delete};

    Router::new()
        // Auth
        .route("/auth/register", post(crate::routes::auth::register))
        .route("/auth/login", post(crate::routes::auth::login))
        .route("/auth/refresh", post(crate::routes::auth::refresh))
        .route("/auth/logout", post(crate::routes::auth::logout))
        .route("/auth/me", get(crate::routes::auth::me))
        // Posts
        .route("/posts/:slug/stats", get(crate::routes::posts::get_stats))
        .route("/posts/:slug/view", post(crate::routes::posts::view))
        .route("/posts/:slug/like", post(crate::routes::posts::like))
        .route("/posts/:slug/like", delete(crate::routes::posts::unlike))
        // Comments
        .route("/posts/:slug/comments", get(crate::routes::comments::list_comments))
        .route("/posts/:slug/comments", post(crate::routes::comments::create_comment))
```

### 4.4 Redis 原子限流实现

```rust
// crates/shared/src/middleware/rate_limit.rs
use axum::{
    extract::{Request, State},
    http::StatusCode,
    middleware::Next,
    response::Response,
};
use std::time::Duration;

// Lua 脚本：固定窗口限流
const RATE_LIMIT_SCRIPT: &str = r#"
local key = KEYS[1]
local limit = tonumber(ARGV[1])
local window = tonumber(ARGV[2])
local current = redis.call("INCR", key)
if current == 1 then
    redis.call("EXPIRE", key, window)
end
if current > limit then
    return 0
end
return 1
"#;

pub async fn rate_limit_middleware<B>(
    State(state): State<crate::api::AppState>,
    request: Request<B>,
    next: Next<B>,
) -> Result<Response, StatusCode> {
    let ip = extract_real_ip(&request);
    let route = extract_route(&request);

    // 构建限流键：rl:{ip}:{route}:{minute_bucket}
    let minute_bucket = chrono::Utc::now().format("%Y%m%d%H%M");
    let key = format!("rl:{}:{}:{}", ip, route, minute_bucket);

    // 根据路由设置不同的限流策略
    let (limit, window) = match route.as_str() {
        "/v1/auth/login" | "/v1/auth/register" => (5, 60), // 5次/分钟
        "/v1/posts" if request.method().as_str() == "POST" => (10, 60), // 创建文章
        "/v1/posts" if request.method().as_str() == "POST" => (100, 60), // 浏览
        _ => (1000, 60), // 默认
    };

    // 执行 Lua 脚本
    let mut conn = state.redis.get().await?;
    let result: i32 = redis::Script::new(RATE_LIMIT_SCRIPT)
        .key(&key)
        .arg(limit)
        .arg(window)
        .invoke_async(&mut conn)
        .await?;

    if result == 0 {
        return Err(StatusCode::TOO_MANY_REQUESTS);
    }

    Ok(next.run(request).await)
}

fn extract_route<B>(request: &Request<B>) -> String {
    // 获取路由模式而不是实际路径
    request
        .extensions()
        .get::<axum::routing::Route>()
        .and_then(|r| Some(r.path().to_string()))
        .unwrap_or_else(|| "unknown".to_string())
}
```

### 4.5 Worker 实现（异步任务处理）

```rust
// crates/worker/src/main.rs
use sqlx::PgPool;
use serde_json::Value;
use std::time::Duration;

#[tokio::main]
async fn main() -> anyhow::Result<()> {
    let settings = crate::shared::config::Settings::from_env()?;
    let db = sqlx::PgPool::connect(&settings.database_url).await?;
    let redis = deadpool_redis::Config::from_url(&settings.redis_url)
        .create_pool(Some(deadpool_redis::Runtime::Tokio1))?;

    tracing::info!("Worker started");

    loop {
        // 批量处理 Outbox 事件
        process_outbox_events(&db, &redis).await?;

        // 批量刷新浏览量
        flush_view_counts(&db, &redis).await?;

        // 等待下次执行
        tokio::time::sleep(Duration::from_secs(5)).await;
    }
}

async fn process_outbox_events(db: &PgPool, redis: &deadpool_redis::Pool) -> anyhow::Result<()> {
    let events = sqlx::query_as!(
        crate::db::models::OutboxEvent,
        r#"
        UPDATE outbox_events
        SET status = 'processing'
        WHERE id = (
            SELECT id FROM outbox_events
            WHERE status = 'pending'
                AND run_after <= NOW()
            ORDER BY created_at
            FOR UPDATE SKIP LOCKED
            LIMIT 100
        )
        RETURNING *
        "#
    )
    .fetch_all(db)
    .await?;

    for event in events {
        match event.topic.as_str() {
            "comment.created" => handle_comment_created(db, &event.payload).await?,
            "post.viewed" => handle_post_viewed(redis, &event.payload).await?,
            _ => tracing::warn!("Unknown event topic: {}", event.topic),
        }

        // 标记为完成
        sqlx::query!(
            "UPDATE outbox_events SET status = 'done' WHERE id = $1",
            event.id
        )
        .execute(db)
        .await?;
    }

    Ok(())
}

async fn flush_view_counts(db: &PgPool, redis: &deadpool_redis::Pool) -> anyhow::Result<()> {
    // 获取所有需要更新的计数
    let mut conn = redis.get().await?;
    let keys: Vec<String> = redis::cmd("KEYS")
        .arg("pv:*")
        .query_async(&mut conn)
        .await?;

    if keys.is_empty() {
        return Ok(());
    }

    // 批量获取计数
    let values: Vec<i64> = redis::cmd("MGET")
        .arg(&keys)
        .query_async(&mut conn)
        .await?;

    // 构建批量更新
    let mut updates = Vec::new();
    for (key, value) in keys.iter().zip(values) {
        if let Some(count) = value {
            let slug = key.strip_prefix("pv:").unwrap_or("");
            updates.push((slug, count));
        }
    }

    // 使用 VALUES 子句批量更新
    if !updates.is_empty() {
        let mut query = String::from(
            r#"
            UPDATE post_stats
            SET view_count = view_count + update_data.count
            FROM (VALUES
            "#
        );

        for (i, (slug, count)) in updates.iter().enumerate() {
            if i > 0 {
                query.push_str(", ");
            }
            query.push_str(&format!("('{}', {})", slug, count));
        }

        query.push_str(
            r#"
            ) AS update_data(slug, count)
            WHERE post_stats.post_slug = update_data.slug
            "#
        );

        sqlx::query(&query).execute(db).await?;

        // 清理已处理的计数
        let _: () = redis::cmd("DEL")
            .arg(&keys)
            .query_async(&mut conn)
            .await?;
    }

    Ok(())
}
```

### 6.1 Next.js 集成

```typescript
// lib/api.ts
import axios from 'axios'
import { getCookie, deleteCookie } from 'cookies-next'

const API_BASE_URL = process.env.NEXT_PUBLIC_API_URL || 'https://api.yourdomain.com/v1'

class ApiClient {
  private axios: AxiosInstance

  constructor() {
    this.axios = axios.create({
      baseURL: API_BASE_URL,
      timeout: 10000,
    })

    // Request interceptor
    this.axios.interceptors.request.use((config) => {
      const token = window.localStorage.getItem('access_token')
      if (token) {
        config.headers.Authorization = `Bearer ${token}`
      }
      return config
    })

    // Response interceptor
    this.axios.interceptors.response.use(
      (response) => response,
      async (error) => {
        if (error.response?.status === 401) {
          // Try to refresh token
          try {
            await this.refreshToken()
            // Retry the original request
            return this.axios.request(error.config)
          } catch {
            // Refresh failed, redirect to login
            window.localStorage.removeItem('access_token')
            window.location.href = '/login'
          }
        }
        return Promise.reject(error)
      }
    )
  }

  private async refreshToken() {
    const response = await axios.post(
      `${API_BASE_URL}/auth/refresh`,
      {},
      {
        withCredentials: true,
      }
    )
    const { access_token } = response.data
    window.localStorage.setItem('access_token', access_token)
  }

  // Auth
  async register(data: RegisterRequest) {
    return this.axios.post('/auth/register', data)
  }

  async login(data: LoginRequest) {
    const response = await this.axios.post('/auth/login', data, {
      withCredentials: true,
    })
    return response.data
  }

  async logout() {
    await this.axios.post('/auth/logout', {}, { withCredentials: true })
    window.localStorage.removeItem('access_token')
    deleteCookie('refresh_token')
  }

  // Posts
  async getPostStats(slug: string) {
    return this.axios.get(`/posts/${slug}/stats`)
  }

  async viewPost(slug: string) {
    return this.axios.post(`/posts/${slug}/view`)
  }

  async likePost(slug: string) {
    return this.axios.post(`/posts/${slug}/like`)
  }

  async unlikePost(slug: string) {
    return this.axios.delete(`/posts/${slug}/like`)
  }

  // Comments
  async getComments(slug: string, params?: { cursor?: string; limit?: number }) {
    return this.axios.get(`/posts/${slug}/comments`, { params })
  }

  async createComment(slug: string, data: CreateCommentRequest) {
    return this.axios.post(`/posts/${slug}/comments`, data)
  }
}

export const api = new ApiClient()
```
